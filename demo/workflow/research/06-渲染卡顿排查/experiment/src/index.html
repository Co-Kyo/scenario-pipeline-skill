// ❌ 同步长任务 — 阻塞主线程 >50ms function processData(items) { let result = 0; for (let i = 0; i < items.length; i++) { result += heavyComputation(items[i]); // 200ms 同步计算 } return result; } // → Long Task! 用户点击无响应，INP > 200ms // ✅ 分片执行 — 每片 <5ms function processChunked(items, chunkSize = 100) { let i = 0; function processNext(deadline) { while (i < items.length && deadline.timeRemaining() > 1) { heavyComputation(items[i++]); } if (i < items.length) requestIdleCallback(processNext); } requestIdleCallback(processNext); } // ✅ Web Worker — 完全不阻塞主线程 const worker = new Worker('compute.js'); worker.postMessage(items); worker.onmessage = (e) => updateUI(e.data);
